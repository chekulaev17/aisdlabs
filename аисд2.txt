template <typename T>
struct Node {
    T data;
    Node* next;
    Node(T value) : data(value), next(nullptr) {}
};

template <typename T>
class LinkedList {
private:
    Node<T>* head;
    Node<T>* tail;
    size_t size;
 void copy_list(const LinkedList& other) {
        if (other.size == 0) return;
        Node<T>* current = other.head;
        do {
            push_tail(current->data);
            current = current->next;
        } while (current != other.head);
    }

    void clear_list() {
        while (size > 0) {
            pop_head();
        }
    }
